不同情况下的JSON数据格式定义如下：

   用户登录请求：

   {
   ​	"op": 1,
   ​	"args": 
   ​		{
   ​			"uname": "dengjun",
   ​			"passwd": "A24314FBECDFDASFD4143241",   # 真实密码的MD5值，使用大写表示
   ​		}
   }
   用户登录响应：
   {
   ​	"op": 1,
   ​	"error_code": 0  # 0表示登录成功，1表示登录失败
   }
   用户注册请求：
   {
   ​	"op": 2,
   ​	"args": 
   ​		{
   ​			"uname": "dengjun",
   ​			"passwd": "A24314FBECDFDASFD4143241",   # 真实密码的MD5值，使用大写表示

   ​			"emai": "dj@itmojun.com"  # 邮箱
   ​		}
   }
   用户注册响应：
   {
   ​	"op": 2,
   ​	"error_code": 0  # 0表示注册成功，1表示注册失败
   }
   
   客户端校验用户名是否存在请求：
   {
   ​	"op": 3,
   ​	"args": 
   ​		{
   ​			"uname": "dengjun"
   ​		}
   }

   }
